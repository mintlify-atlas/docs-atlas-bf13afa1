This quickstart walks you through building a complete Watermill application from scratch. You will create a Router with a real handler, publish structured events, and process them — all using the built-in `GoChannel` pub/sub so there is nothing to install beyond Go itself. By the end you will understand how messages flow through a Watermill application and how Ack/Nack signals control redelivery.

The finished code is available at [`_examples/basic/1-your-first-app/main.go`](https://github.com/ThreeDotsLabs/watermill/blob/master/_examples/basic/1-your-first-app/main.go) on GitHub.

## Prerequisites

- Go 1.21 or later
- Basic familiarity with Go modules and goroutines

## Build the app

<Steps>
  <Step title="Install Watermill">
    Create a new module and fetch Watermill:

    ```bash
    mkdir watermill-app && cd watermill-app
    go mod init watermill-app
    go get github.com/ThreeDotsLabs/watermill
    ```
  </Step>

  <Step title="Define your message types">
    Create `main.go` and define the structs for the events this app will produce and consume. Using typed structs with JSON keeps payloads readable and safe.

    ```go
    package main

    import (
        "context"
        "encoding/json"
        "fmt"
        "time"

        "github.com/ThreeDotsLabs/watermill"
        "github.com/ThreeDotsLabs/watermill/message"
        "github.com/ThreeDotsLabs/watermill/message/router/middleware"
        "github.com/ThreeDotsLabs/watermill/message/router/plugin"
        "github.com/ThreeDotsLabs/watermill/pubsub/gochannel"
    )

    var (
        consumeTopic = "events"
        publishTopic = "events-processed"

        logger = watermill.NewStdLogger(
            true,  // debug
            false, // trace
        )
    )

    type event struct {
        ID int `json:"id"`
    }

    type processedEvent struct {
        ProcessedID int       `json:"processed_id"`
        Time        time.Time `json:"time"`
    }
    ```
  </Step>

  <Step title="Create the Router">
    The Router is the central piece of a Watermill application. It wires subscribers to handlers, manages goroutines, and handles graceful shutdown. `plugin.SignalsHandler` shuts it down cleanly when `SIGTERM` is received. `middleware.Recoverer` converts panics into errors so they can be handled by retry logic.

    ```go
    func main() {
        pubSub := gochannel.NewGoChannel(gochannel.Config{}, logger)

        router, err := message.NewRouter(message.RouterConfig{}, logger)
        if err != nil {
            panic(err)
        }

        router.AddPlugin(plugin.SignalsHandler)
        router.AddMiddleware(middleware.Recoverer)
    ```
  </Step>

  <Step title="Register a handler">
    A handler binds an input topic and subscriber to an output topic and publisher, with a function that transforms messages. The function signature `func(*message.Message) ([]*message.Message, error)` is the core contract: return new messages to publish, or return an error to signal a processing failure.

    ```go
        // Adding a handler (multiple handlers can be added)
        router.AddHandler(
            "handler_1",  // handler name, must be unique
            consumeTopic, // topic from which messages should be consumed
            pubSub,
            publishTopic, // topic to which messages should be published
            pubSub,
            func(msg *message.Message) ([]*message.Message, error) {
                consumedPayload := event{}
                err := json.Unmarshal(msg.Payload, &consumedPayload)
                if err != nil {
                    // When a handler returns an error, the default behavior is to send a Nack
                    // (negative-acknowledgement). The message will be processed again.
                    //
                    // You can change the default behaviour by using middlewares, like Retry or PoisonQueue.
                    // You can also implement your own middleware.
                    return nil, err
                }

                fmt.Printf("received event %+v\n", consumedPayload)

                newPayload, err := json.Marshal(processedEvent{
                    ProcessedID: consumedPayload.ID,
                    Time:        time.Now(),
                })
                if err != nil {
                    return nil, err
                }

                newMessage := message.NewMessage(watermill.NewUUID(), newPayload)

                return []*message.Message{newMessage}, nil
            },
        )
    ```

    <Note>
      The example uses one `pubSub` value for both the subscriber and publisher because `GoChannel` implements both interfaces. In production you would typically pass a dedicated subscriber and publisher backed by different (or the same) broker instances.
    </Note>
  </Step>

  <Step title="Simulate incoming events">
    In a real application, events arrive from an external system. Here a goroutine publishes one event per second to simulate incoming traffic. Each message gets a UUID (useful for tracing) and a JSON-encoded payload.

    ```go
        // Simulate incoming events in the background
        go simulateEvents(pubSub)

        if err := router.Run(context.Background()); err != nil {
            panic(err)
        }
    }

    // simulateEvents produces events that will be later consumed.
    func simulateEvents(publisher message.Publisher) {
        i := 0
        for {
            e := event{
                ID: i,
            }

            payload, err := json.Marshal(e)
            if err != nil {
                panic(err)
            }

            err = publisher.Publish(consumeTopic, message.NewMessage(
                watermill.NewUUID(), // internal uuid of the message, useful for debugging
                payload,
            ))
            if err != nil {
                panic(err)
            }

            i++

            time.Sleep(time.Second)
        }
    }
    ```
  </Step>

  <Step title="Run the app">
    ```bash
    go run main.go
    ```

    You should see log output similar to:

    ```
    received event {ID:0}
    received event {ID:1}
    received event {ID:2}
    ```

    Press `Ctrl+C` to trigger a graceful shutdown via `SIGTERM`.
  </Step>
</Steps>

## Understanding Ack and Nack

Every message received from a `Subscriber` must be either acknowledged or negatively acknowledged. This is how Watermill (and the underlying Pub/Sub) knows whether to deliver the next message.

| Call | Meaning | What happens next |
|---|---|---|
| `msg.Ack()` | Processing succeeded. | The next message is delivered. |
| `msg.Nack()` | Processing failed. | The message is redelivered (behavior depends on the Pub/Sub). |

When using the Router, you rarely call `Ack` or `Nack` directly. The Router calls `Ack` automatically when your handler returns `nil`, and `Nack` when it returns an error.

```go
// Ack is sent automatically when the handler returns nil.
return []*message.Message{newMessage}, nil

// Nack is sent automatically when the handler returns an error.
// The message will be redelivered.
return nil, fmt.Errorf("failed to process message: %w", err)
```

<Tip>
  Use middleware like `middleware.Retry` to automatically retry failed messages a configurable number of times before giving up, and `middleware.PoisonQueue` to route permanently-failing messages to a dead-letter topic.
</Tip>

## Complete source

Here is the full `main.go` using GoChannel in place of Kafka, making it runnable with no external dependencies:

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "time"

    "github.com/ThreeDotsLabs/watermill"
    "github.com/ThreeDotsLabs/watermill/message"
    "github.com/ThreeDotsLabs/watermill/message/router/middleware"
    "github.com/ThreeDotsLabs/watermill/message/router/plugin"
    "github.com/ThreeDotsLabs/watermill/pubsub/gochannel"
)

var (
    consumeTopic = "events"
    publishTopic = "events-processed"

    logger = watermill.NewStdLogger(
        true,  // debug
        false, // trace
    )
)

type event struct {
    ID int `json:"id"`
}

type processedEvent struct {
    ProcessedID int       `json:"processed_id"`
    Time        time.Time `json:"time"`
}

func main() {
    pubSub := gochannel.NewGoChannel(gochannel.Config{}, logger)

    router, err := message.NewRouter(message.RouterConfig{}, logger)
    if err != nil {
        panic(err)
    }

    router.AddPlugin(plugin.SignalsHandler)
    router.AddMiddleware(middleware.Recoverer)

    router.AddHandler(
        "handler_1",
        consumeTopic,
        pubSub,
        publishTopic,
        pubSub,
        func(msg *message.Message) ([]*message.Message, error) {
            consumedPayload := event{}
            err := json.Unmarshal(msg.Payload, &consumedPayload)
            if err != nil {
                return nil, err
            }

            fmt.Printf("received event %+v\n", consumedPayload)

            newPayload, err := json.Marshal(processedEvent{
                ProcessedID: consumedPayload.ID,
                Time:        time.Now(),
            })
            if err != nil {
                return nil, err
            }

            newMessage := message.NewMessage(watermill.NewUUID(), newPayload)
            return []*message.Message{newMessage}, nil
        },
    )

    go simulateEvents(pubSub)

    if err := router.Run(context.Background()); err != nil {
        panic(err)
    }
}

func simulateEvents(publisher message.Publisher) {
    i := 0
    for {
        e := event{ID: i}

        payload, err := json.Marshal(e)
        if err != nil {
            panic(err)
        }

        err = publisher.Publish(consumeTopic, message.NewMessage(
            watermill.NewUUID(),
            payload,
        ))
        if err != nil {
            panic(err)
        }

        i++
        time.Sleep(time.Second)
    }
}
```
